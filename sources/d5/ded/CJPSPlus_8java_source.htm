55  @Override
56  public final List<DoubleMatrix1D> route( final ObjectMatrix2D p_objects, final DoubleMatrix1D p_currentposition, final DoubleMatrix1D p_targetposition )
57  {
58  final TreeSet<CJumpPoint> l_openlist = new TreeSet<>( new CCompareJumpPoint() );
59  final ArrayList<DoubleMatrix1D> l_closedlist = new ArrayList<>();
60  final List<DoubleMatrix1D> l_finalpath = new ArrayList<>();
61 
62  l_openlist.add( new CJumpPoint( p_currentposition, null ) );
63  while ( !l_openlist.isEmpty() )
64  {
65  final CJumpPoint l_currentnode = l_openlist.pollFirst();
66  //if the current node is the end node
67  if ( l_currentnode.coordinate().equals( p_targetposition ) )
68  {
69  l_finalpath.add( p_targetposition );
70  CJumpPoint l_parent = l_currentnode.parent();
71  while ( !l_parent.coordinate().equals( p_currentposition ) )
72  {
73  l_finalpath.add( l_parent.coordinate() );
74  l_parent = l_parent.parent();
75  }
76  Collections.reverse( l_finalpath );
77  return l_finalpath;
78  }
79  //Find the successors to current node (add them to the open list)
80  this.successors( p_objects, l_currentnode, p_targetposition, l_closedlist, l_openlist );
81 
82  //Add the l_currentnode to the closed list (as to not open it again)
83  l_closedlist.add( l_currentnode.coordinate() );
84 
85  }
86 
87  return Collections.<DoubleMatrix1D>emptyList();
88  }

90  @Override
91  public final double estimatedtime( final Stream<DoubleMatrix1D> p_route, final double p_speed )
92  {
93  return StreamUtils.windowed( p_route, 2, 1 )
94  .mapToDouble( i -> Math.sqrt(
95  Algebra.DEFAULT.norm2(
96  new DenseDoubleMatrix1D( i.get( 1 ).toArray() )
97  .assign( i.get( 0 ), Functions.minus )
98  )
99  ) / p_speed
100  )
101  .sum();
102  }

113  private void successors( final ObjectMatrix2D p_objects, final CJumpPoint p_curnode, final DoubleMatrix1D p_target,
114  final ArrayList<DoubleMatrix1D> p_closedlist,
115  final Set<CJumpPoint> p_openlist
116  )
117  {
118  IntStream.rangeClosed( -1, 1 )
119  .forEach( i -> IntStream.rangeClosed( -1, 1 )
120  .filter( j -> ( i != 0 || j != 0 )
121  && !this.isNotNeighbour( p_objects, p_curnode.coordinate().getQuick( 0 ) + i, p_curnode.coordinate().getQuick( 1 ) + j, p_closedlist )
122  && !this.isOccupied( p_objects, p_curnode.coordinate().getQuick( 0 ) + i, p_curnode.coordinate().getQuick( 1 ) + j ) )
123  .forEach( j ->
124  {
125  final DoubleMatrix1D l_nextjumpnode = this.jump( p_curnode.coordinate(), p_target, i, j, p_objects );
126  this.addsuccessors( l_nextjumpnode, p_closedlist, p_openlist, p_curnode, p_target );
127  } )
128  );
129  }

140  private void addsuccessors( final DoubleMatrix1D p_nextjumpnode, final ArrayList<DoubleMatrix1D> p_closedlist,
141  final Set<CJumpPoint> p_openlist, final CJumpPoint p_curnode, final DoubleMatrix1D p_target
142  )
143  {
144  if ( !( p_nextjumpnode != null && !p_curnode.coordinate().equals( p_nextjumpnode ) && ( !p_closedlist.contains( p_nextjumpnode ) ) ) )
145  return;
146 
147  final CJumpPoint l_jumpnode = new CJumpPoint( p_nextjumpnode, p_curnode );
148  this.calculateScore( l_jumpnode, p_target );
149  p_openlist.removeIf( s -> s.coordinate().equals( p_nextjumpnode ) && s.fscore() > l_jumpnode.fscore() );
150 
151  //checking that the jump point is already exists in open list or not, if yes then check their fscore to make decision
152  if ( !p_openlist.parallelStream().filter( s -> s.coordinate().equals( p_nextjumpnode ) ).anyMatch( s -> s.fscore() < l_jumpnode.fscore() ) )
153  p_openlist.add( l_jumpnode );
154 
155  }

167  private DoubleMatrix1D jump( final DoubleMatrix1D p_curnode, final DoubleMatrix1D p_target,
168  final double p_row, final double p_col, final ObjectMatrix2D p_objects
169  )
170  {
171  //The next nodes details
172  final double l_nextrow = p_curnode.getQuick( 0 ) + p_row;
173  final double l_nextcol = p_curnode.getQuick( 1 ) + p_col;
174  final DoubleMatrix1D l_nextnode = new DenseDoubleMatrix1D( new double[]{l_nextrow, l_nextcol} );
175 
176  // If the l_nextnode is outside the grid or occupied then return null
177  if ( this.isOccupied( p_objects, l_nextrow, l_nextcol ) || this.isNotCoordinate( p_objects, l_nextrow, l_nextcol ) )
178  return null;
179 
180  //If the l_nextnode is the target node then return it
181  if ( p_target.equals( l_nextnode ) )
182  return l_nextnode;
183 
184  //If we are going in a diagonal direction check for forced neighbors
185  if ( p_row != 0 && p_col != 0 )
186  {
187  final DoubleMatrix1D l_node = this.diagjump( l_nextrow, l_nextcol, l_nextnode, p_target, p_row, p_col, p_objects );
188  if ( l_node != null )
189  return l_node;
190  }
191  else if ( this.horizontal( l_nextrow, l_nextcol, p_row, p_objects, 1 ) || this.horizontal( l_nextrow, l_nextcol, p_row, p_objects, -1 )
192  || this.vertical( l_nextrow, l_nextcol, p_col, p_objects, 1 ) || this.vertical( l_nextrow, l_nextcol, p_col, p_objects, -1 ) )
193  return l_nextnode;
194 
195  return this.jump( l_nextnode, p_target, p_row, p_col, p_objects );
196  }

210  private DoubleMatrix1D diagjump( final double p_nextrow, final double p_nextcolumn, final DoubleMatrix1D p_nextnode, final DoubleMatrix1D p_target,
211  final double p_row, final double p_col, final ObjectMatrix2D p_objects
212  )
213  {
214  if ( this.diagonal( p_nextrow, p_nextcolumn, -p_row, p_col, p_row, 0, p_objects ) || this.diagonal(
215  p_nextrow, p_nextcolumn, p_row, -p_col, 0, p_col, p_objects ) )
216  return p_nextnode;
217 
218  //before each diagonal step the algorithm must first fail to detect any straight jump points
219  if ( this.jump( p_nextnode, p_target, p_row, 0, p_objects ) != null || this.jump( p_nextnode, p_target, 0, p_col, p_objects ) != null )
220  return p_nextnode;
221 
222  return null;
223  }

327  private void calculateScore( final CJumpPoint p_jumpnode, final DoubleMatrix1D p_target )
328  {
329  final double l_hscore = Math.abs( p_target.getQuick( 0 ) - p_jumpnode.coordinate().getQuick( 0 ) ) * 10
330  + Math.abs( p_target.getQuick( 1 ) - p_jumpnode.coordinate().getQuick( 1 ) ) * 10;
331 
332  p_jumpnode.hscore( l_hscore );
333 
334  final double l_row = Math.abs( p_jumpnode.parent().coordinate().getQuick( 0 ) - p_jumpnode.coordinate().getQuick( 0 ) );
335  final double l_column = Math.abs( p_jumpnode.parent().coordinate().getQuick( 1 ) - p_jumpnode.coordinate().getQuick( 1 ) );
336  final double l_gscore = ( l_row != 0 && l_column != 0 ) ? Math.abs( 14 * l_row ) : Math.abs( 10 * Math.max( l_row, l_column ) );
337 
338  p_jumpnode.gscore( p_jumpnode.parent().gscore() + l_gscore );
339  }
