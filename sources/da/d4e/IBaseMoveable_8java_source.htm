120  @Override
121  public void move( final ObjectMatrix2D p_grid, final DoubleMatrix1D p_newposition )
122  {
123  this.cells( this.position() ).forEach( i -> p_grid.setQuick( i.getLeft(), i.getRight(), null ) );
124  this.cells( p_newposition ).forEach( i -> p_grid.setQuick( i.getLeft(), i.getRight(), this ) );
125  this.position().setQuick( 0, p_newposition.getQuick( 0 ) );
126  this.position().setQuick( 1, p_newposition.getQuick( 1 ) );
127  }
