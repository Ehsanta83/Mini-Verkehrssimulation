74  public DoubleMatrix1D position( final DoubleMatrix1D p_position, final DoubleMatrix1D p_goalposition, final int p_speed )
75  {
76  // calculate the stright line by: current position + l * (goal position - current position)
77  // normalize direction and rotate the normalized vector based on the direction
78  // calculate the target position based by: current position + speed * rotate( normalize( goal position - current position ) )
79  final DoubleMatrix1D l_view = new DenseDoubleMatrix1D( p_goalposition.toArray() );
80  return CMath.ALGEBRA.mult(
81  m_rotation,
82  l_view
83  .assign( p_position, Functions.minus )
84  .assign( Functions.div( Math.sqrt( Algebra.DEFAULT.norm2( l_view ) ) ) )
85  )
86  .assign( Functions.mult( p_speed ) )
87  .assign( p_position, Functions.plus )
88  .assign( Math::round );
89  }
