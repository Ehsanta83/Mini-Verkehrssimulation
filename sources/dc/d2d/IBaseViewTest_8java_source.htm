308  private static TiledMap tilemap( final int p_row, final int p_column, final int p_cellsize )
309  {
310  // create background checkerboard with a tile map
311  final Pixmap l_pixmap = new Pixmap( 2 * p_cellsize, p_cellsize, Pixmap.Format.RGBA8888 );
312  l_pixmap.setColor( new Color( 0.8f, 0.1f, 0.1f, 0.5f ) );
313  l_pixmap.fillRectangle( 0, 0, p_cellsize, p_cellsize );
314  l_pixmap.setColor( new Color( 0.5f, 0.5f, 0.5f, 0.5f ) );
315  l_pixmap.fillRectangle( p_cellsize, 0, p_cellsize, p_cellsize );
316 
317  final Texture l_texture = new Texture( l_pixmap );
318  final TiledMapTile l_region1 = new StaticTiledMapTile( new TextureRegion( l_texture, 0, 0, p_cellsize, p_cellsize ) );
319  final TiledMapTile l_region2 = new StaticTiledMapTile( new TextureRegion( l_texture, p_cellsize, 0, p_cellsize, p_cellsize ) );
320 
321  // create tilemap
322  final TiledMap l_map = new TiledMap();
323  final TiledMapTileLayer l_layer = new TiledMapTileLayer( p_column, p_row, p_cellsize, p_cellsize );
324  l_map.getLayers().add( l_layer );
325 
326  IntStream
327  .range( 0, p_column )
328  .forEach( x ->
329  {
330  IntStream
331  .range( 0, p_row )
332  .forEach( y ->
333  {
334  final TiledMapTileLayer.Cell l_cell = new TiledMapTileLayer.Cell();
335  l_layer.setCell( x, y, l_cell );
336  l_cell.setTile(
337  y % 2 != 0
338  ? x % 2 != 0 ? l_region1 : l_region2
339  : x % 2 != 0 ? l_region2 : l_region1
340  );
341  } );
342  } );
343 
344  return l_map;
345  }

348  @Override
349  public final void create()
350  {
351  // create orthogonal camera perspective
352  m_unit = 1.0f / m_cellsize;
353 
354  // create execution structure for painting
355  m_spritebatch = new SpriteBatch();
356 
357  // create environment view and put all objects in it
358  m_render = new OrthogonalTiledMapRenderer( tilemap( m_cellrows, m_cellcolumns, m_cellsize ), m_unit, m_spritebatch );
359 
360  m_camera = new OrthographicCamera( m_cellcolumns, m_cellrows );
361  m_camera.setToOrtho( false, m_cellcolumns * m_unit, m_cellrows * m_unit );
362  m_camera.position.set( m_cellcolumns / 2f, m_cellrows / 2f, 0 );
363  m_camera.zoom = m_cellsize;
364  m_render.setView( m_camera );
365 
366  // set input processor
367  Gdx.input.setInputProcessor( this );
368  }

370  @Override
371  public final void render()
372  {
373  // camera update must be the first for reaction on input device events
374  m_camera.update();
375 
376  // create black background
377  Gdx.gl.glClearColor( 0, 0, 0, 1 );
378  Gdx.gl.glClear( GL20.GL_COLOR_BUFFER_BIT );
379 
380  // environment tilemap painting
381  m_render.setView( m_camera );
382  m_render.render();
383 
384 
385  // object sprite painting
386  m_spritebatch.setProjectionMatrix( m_camera.combined );
387  m_spritebatch.begin();
388 
389  m_sprites.stream()
390  .map( i -> i.spriteinitialize( m_cellrows, m_cellcolumns, m_cellsize, m_unit ) )
391  .filter( Objects::nonNull )
392  .forEach( i -> i.sprite().draw( m_spritebatch ) );
393 
394  m_spritebatch.end();
